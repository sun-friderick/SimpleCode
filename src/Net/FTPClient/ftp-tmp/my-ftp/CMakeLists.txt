########################################################################################
#############              生成 json 库文件                               ############## 
########################################################################################
LIST (APPEND json_LIB_SRCS
    ${CMAKE_CURRENT_SOURCE_DIR}/cJSON.c  
    )


    
    
########################################################################################
#############              添加头文件目录                                 ############## 
########################################################################################
include_directories(  
    ${PROJECT_SOURCE_DIR}/includes  
    ${PROJECT_SOURCE_DIR}/utils
    ${PROJECT_SOURCE_DIR}/utils/JsonC
)  

########################################################################################
#############               添加 链接库文件目录                           ############## 
########################################################################################
link_directories(  
    /usr/local/lib   
    )  


    
########################################################################################
#############           生成目标库文件 并 标明版本号                      ############## 
########################################################################################
IF (MODULE_json)
    #生成动态库  静态库则 STATIC  
    add_library (jsonlib SHARED      ${json_LIB_SRCS})  
    add_library (jsons STATIC        ${json_LIB_SRCS})  
    # 设置版本号，SOVERSION为API版本号
    set_target_properties(jsonlib   PROPERTIES VERSION 1.0.0 SOVERSION 1 )
ELSE (MODULE_json)
    MESSAGE(STATUS "Not Include json module.")
ENDIF (MODULE_json)
